🔧 Google Sheets Setup

1Create your Google Sheet

Create a new Google Sheet with two tabs named exactly:

  • Players — columns: Name, Batch, Phone
  • Scores — columns: Name, Match, Points, Date
2Create an Apps Script
  1. In your Sheet, go to Extensions → Apps Script
  2. Delete any existing code and paste the code below
  3. Click Save, then Deploy → New deployment
  4. Choose type: Web app, set access to Anyone
  5. Copy the deployment URL
// IPL Leaderboard - Google Apps Script Backend const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); function doGet(e) { const action = e.parameter.action; if (action === 'getPlayers') return getPlayers(); if (action === 'getScores') return getScores(); if (action === 'getAll') return getAll(); return jsonResponse({error: 'Unknown action'}); } function doPost(e) { const data = JSON.parse(e.postData.contents); if (data.action === 'addPlayer') return addPlayer(data); if (data.action === 'updateScore') return updateScore(data); if (data.action === 'addMatch') return addMatch(data); if (data.action === 'bulkUpdateScores') return bulkUpdateScores(data); return jsonResponse({error: 'Unknown action'}); } function getAll() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const players = sheetToArray(ss.getSheetByName('Players')); const scores = sheetToArray(ss.getSheetByName('Scores')); return jsonResponse({players, scores}); } function getPlayers() { const ss = SpreadsheetApp.getActiveSpreadsheet(); return jsonResponse(sheetToArray(ss.getSheetByName('Players'))); } function getScores() { const ss = SpreadsheetApp.getActiveSpreadsheet(); return jsonResponse(sheetToArray(ss.getSheetByName('Scores'))); } function addPlayer(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('Players'); sheet.appendRow([data.name, data.batch, data.phone || '']); return jsonResponse({success: true}); } function updateScore(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('Scores'); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.name && rows[i][1] === data.match) { sheet.getRange(i + 1, 3).setValue(data.points); return jsonResponse({success: true, updated: true}); } } sheet.appendRow([data.name, data.match, data.points, new Date().toISOString()]); return jsonResponse({success: true, inserted: true}); } function bulkUpdateScores(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('Scores'); data.scores.forEach(s => { const rows = sheet.getDataRange().getValues(); let found = false; for (let i = 1; i < rows.length; i++) { if (rows[i][0] === s.name && rows[i][1] === s.match) { sheet.getRange(i + 1, 3).setValue(s.points); found = true; break; } } if (!found) sheet.appendRow([s.name, s.match, s.points, new Date().toISOString()]); }); return jsonResponse({success: true}); } function addMatch(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('Matches'); if (!sheet) { sheet = ss.insertSheet('Matches'); sheet.appendRow(['MatchNum','Name','Date']); } sheet.appendRow([data.matchNum, data.name, data.date]); return jsonResponse({success: true}); } function sheetToArray(sheet) { if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
